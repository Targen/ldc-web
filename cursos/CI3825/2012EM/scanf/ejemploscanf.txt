#include <stdio.h>
#define MAXNOMBRE 100


int main(int argc, char *argv[]) 
 {
     char nom[MAXNOMBRE];
     FILE *fp;
     int i, k, edad=0, tiempo=0;


     /* Una forma de leer */
     fp = fopen(argv[1], "r");
     i = 0;

     while (!feof(fp)) {
       fscanf(fp, "%d,%d,%s", &edad, &tiempo, nom);
       printf("%d, %d, %s\n", edad, tiempo, nom);
           
     }
     
  
      fclose(fp); 
      return(0); 



}
